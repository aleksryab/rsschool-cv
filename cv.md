function firstNonRepeatingLetter(s) {
  for (const char of s) {
    let counter = 0;
    for (const it of s) {
      if (it.toLowerCase() === char.toLowerCase()) counter ++;
    }
    if (counter === 1) return char;
  }
  return '';
}
